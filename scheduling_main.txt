    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int choice;

        do {
            System.out.println("Menu");
            System.out.println("1. Round Robin Scheduling");
            System.out.println("2. Shortest Job First Scheduling");
            System.out.println("3. Preemptive Priority Scheduling");
            System.out.println("4. Exit");
            System.out.print("Enter your choice: ");
            choice = scanner.nextInt();

            switch (choice) {
                case 1:
                    System.out.print("Enter the number of processes: ");
                    int n1 = scanner.nextInt();
                    int process1[] = new int[n1];
                    int burst_time1[] = new int[n1];
                    int arrival_time1[] = new int[n1];
                    int quantum1 = 1;

                    for (int i = 0; i < n1; i++) {
                        System.out.print("Enter arrival time for process " + (i + 1) + ": ");
                        arrival_time1[i] = scanner.nextInt();

                        System.out.print("Enter burst time for process " + (i + 1) + ": ");
                        burst_time1[i] = scanner.nextInt();

                        process1[i] = i + 1;
                    }

                    roundRobinScheduling(process1, n1, burst_time1, arrival_time1, quantum1);
                    break;

                case 2:
                    shortestJobFirstScheduling();
                    break;

                case 3:
                    preemptivePriorityScheduling();
                    break;

                case 4:
                    System.out.println("Exiting the program...");
                    break;

                default:
                    System.out.println("Invalid choice. Please try again.");
            }
            System.out.println();
        } while (choice != 4);

        scanner.close();
    }
}
